create table receta (IDRECETA INTEGER PRIMARY KEY  AUTOINCREMENT ,  nombre varchar(100))
create table ingrediente (IDINGREDIENTE INTEGER PRIMARY KEY  AUTOINCREMENT ,  nombre varchar(100))
create table receta_ingrediente(id INTEGER primary key autoincrement, IDINGREDIENTE INTEGER, IDRECETA INTEGER, CANTIDAD INTEGER NOT NULL, FOREIGN KEY(IDINGREDIENTE) REFERENCES ingrediente(IDINGREDIENTE), FOREIGN KEY (IDRECETA) REFERENCES receta(IDRECETA))
